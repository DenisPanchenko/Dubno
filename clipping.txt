uniform sampler2D quadTexture;
varying vec2 vTex;


void main(void)
{   
    vec4 MaxColor = vec4(1.0);
    vec4 MinColor = vec4(0.5,0.5,0.5,1.0);
    vec4 ColorRange = (MaxColor-MinColor);
    vec4 orig = texture2D(quadTexture, vTex);
    orig = min(MaxColor,orig);
    orig = max(MinColor,orig);
    orig = (orig-MinColor)/ColorRange;
    gl_FragColor = orig;
}
